C# Code:⌗ 

using System;
using System.Collections;
using System.Collections.Generic;
using System.IO;
using Microsoft.VisualBasic.CompilerServices; // Install-Package Microsoft.VisualBasic
using WordCloud;

public partial class Form1
{
    public Form1()
    {
        base.Load += Form1_Load;
    }

    private object text;
    private Dictionary<string, int> words = new Dictionary<string, int>();

    private void Form1_Load(object sender, EventArgs e)
    {
        this.RichTextBox1.EnableAutoDragDrop = true;
    }

    private void RichTextBox1_DragDrop(object sender, Global.System.Windows.Forms.DragEventArgs e)
    {
        e.Effect = DragDropEffects.None;
        this.RichTextBox1.Clear();
        string[] files = (string[])e.Data.GetData(DataFormats.FileDrop);
        foreach (var path in files)
            ProcessaFile(path);
    }

    private void RichTextBox1_DragEnter(object sender, Global.System.Windows.Forms.DragEventArgs e)
    {
        if (e.Data.GetDataPresent(DataFormats.FileDrop))
        {
            e.Effect = DragDropEffects.Copy;
        }
    }

    public void ProcessaFile(string path)
    {
        this.RichTextBox1.AppendText(path + Environment.NewLine);
        Button2.Enabled = true;
    }

    private void ButtonGetFile_Click(object sender, EventArgs e)
    {
        var ofd = new OpenFileDialog();
        ofd.ShowDialog();
        if (ofd.FileName is object)
        {
            this.RichTextBox1.Text = ofd.FileName;
            Button2.Enabled = true;
        }
    }

    private void Button2_Click(object sender, EventArgs e)
    {
        this.RichTextBox2.Clear();
        string path = this.RichTextBox1.Text;
        var sr = new StreamReader(path);
        do
        {
            string line = sr.ReadLine();
            if (sr.EndOfStream)
            {
                break;
            }

            this.RichTextBox2.AppendText(line + Environment.NewLine);
        }
        while (true);
        text = File.ReadAllText(path).Replace(",", "").ToLower();
        var wordsInFile = text.Split(' ');
        foreach (var word in (IEnumerable)wordsInFile)
        {
            if (words.ContainsKey(Conversions.ToString(word)))
            {
                words[Conversions.ToString(word)] += 1;
            }
            else
            {
                words[Conversions.ToString(word)] = 1;
            }
        }

        this.RichTextBox2.AppendText(Environment.NewLine + Environment.NewLine);
        this.RichTextBox2.AppendText("FREQUENCY OF WORDS IN TEXT" + Environment.NewLine);
        this.RichTextBox2.AppendText("___________________________________" + Environment.NewLine);
        this.RichTextBox2.AppendText("Word".PadRight(30) + "Frequency".PadRight(30) + Environment.NewLine);
        Button3.Enabled = true;
    }

    private void Button3_Click(object sender, EventArgs e)
    {
        var wordss = new List<string>();
        var frequency = new List<int>();
        foreach (KeyValuePair<string, int> kvp in words)
        {
            wordss.Add(kvp.Key.ToString());
            frequency.Add(kvp.Value);
            this.RichTextBox2.AppendText(kvp.Key.ToString().PadRight(30) + " " + kvp.Value.ToString().PadRight(30) + Environment.NewLine);
        }

        var wc = new WordCloud.WordCloud(480, 360, true);
        wc.Draw(wordss, frequency).Save(@"C:\Users\Luca\Desktop\WordCloud.JPG");
        string imagePath = @"C:\Users\Luca\Desktop\WordCloud.JPG";
        this.PictureBox1.Image = new Bitmap(imagePath);
    }
}

VB.NET Code:⌗ 

Imports System.IO
Imports WordCloud

Public Class Form1
    Dim text
    Dim words As Dictionary(Of String, Integer) = New Dictionary(Of String, Integer)()
    Private Sub Form1_Load(sender As Object, e As EventArgs) Handles MyBase.Load
        Me.RichTextBox1.EnableAutoDragDrop = True
    End Sub
    Private Sub RichTextBox1_DragDrop(ByVal sender As Object, ByVal e As System.Windows.Forms.DragEventArgs) Handles RichTextBox1.DragDrop
        e.Effect = DragDropEffects.None

        Me.RichTextBox1.Clear()
        Dim files() As String = CType(e.Data.GetData(DataFormats.FileDrop), String())

        For Each path In files
            Me.ProcessaFile(path)
        Next
    End Sub
    Private Sub RichTextBox1_DragEnter(ByVal sender As Object, ByVal e As System.Windows.Forms.DragEventArgs) Handles RichTextBox1.DragEnter
        If e.Data.GetDataPresent(DataFormats.FileDrop) Then
            e.Effect = DragDropEffects.Copy
        End If
    End Sub
    Sub ProcessaFile(path As String)
        Me.RichTextBox1.AppendText(path & Environment.NewLine)
        Button2.Enabled = True
    End Sub
    Private Sub ButtonGetFile_Click(sender As Object, e As EventArgs) Handles Button1.Click
        Dim ofd As New OpenFileDialog
        ofd.ShowDialog()

        If ofd.FileName IsNot Nothing Then
            Me.RichTextBox1.Text = ofd.FileName
            Button2.Enabled = True
        End If
    End Sub
    Private Sub Button2_Click(sender As Object, e As EventArgs) Handles Button2.Click
        Me.RichTextBox2.Clear()
        Dim path As String = Me.RichTextBox1.Text
        Dim sr As New StreamReader(path)

        Do
            Dim line As String = sr.ReadLine
            If sr.EndOfStream Then
                Exit Do
            End If
            Me.RichTextBox2.AppendText(line & Environment.NewLine)
        Loop

        text = File.ReadAllText(path).Replace(",", "").ToLower()
        Dim wordsInFile = text.Split(" "c)

        For Each word In wordsInFile
            If words.ContainsKey(word) Then
                words(word) += 1
            Else
                words(word) = 1
            End If
        Next
        Me.RichTextBox2.AppendText(Environment.NewLine & Environment.NewLine)
        Me.RichTextBox2.AppendText("FREQUENCY OF WORDS IN TEXT" & Environment.NewLine)
        Me.RichTextBox2.AppendText("___________________________________" & Environment.NewLine)
        Me.RichTextBox2.AppendText("Word".PadRight(30) & "Frequency".PadRight(30) & Environment.NewLine)
        Button3.Enabled = True
    End Sub

    Private Sub Button3_Click(sender As Object, e As EventArgs) Handles Button3.Click
        Dim wordss As New List(Of String)
        Dim frequency As New List(Of Integer)
        For Each kvp As KeyValuePair(Of String, Integer) In words
            wordss.Add(kvp.Key.ToString())
            frequency.Add(kvp.Value)
            Me.RichTextBox2.AppendText(kvp.Key.ToString().PadRight(30) & " " & kvp.Value.ToString().PadRight(30) & Environment.NewLine)
        Next
        Dim wc As New WordCloud.WordCloud(480, 360, True)
        wc.Draw(wordss, frequency).Save("C:\Users\Luca\Desktop\WordCloud.JPG")
        Dim imagePath As String = "C:\Users\Luca\Desktop\WordCloud.JPG"
        Me.PictureBox1.Image = New Bitmap(imagePath)
    End Sub
End Class
