4_A : RAMEN AND STATISTICS

2021-10-09 :: Luca

Intro⌗

In this task I have created a program that takes as input a CSV file regarding some Ramen stores around the world. It is possible to visualize all the data in the file and some interesting statistics are computed, both in the case of one variable and in the case of two variables. If you want to try the program, besides downloading the source code at the end of the page, clicking here you will find the link to download the CSV file.

here, you can see my program in action

C# Code:⌗ 

using System;
using System.Collections.Generic;
using Microsoft.VisualBasic.CompilerServices; // Install-Package Microsoft.VisualBasic
using Microsoft.VisualBasic.FileIO; // Install-Package Microsoft.VisualBasic

public partial class Form1
{
    public Form1()
    {
        base.Load += Form1_Load;
    }

    private List<Ramen> listOfValues = new List<Ramen>();

    private void Form1_Load(object sender, EventArgs e)
    {
        this.RichTextBox1.EnableAutoDragDrop = true;
    }

    private void RichTextBox1_DragDrop(object sender, Global.System.Windows.Forms.DragEventArgs e)
    {
        e.Effect = DragDropEffects.None;
        this.RichTextBox1.Clear();
        string[] files = (string[])e.Data.GetData(DataFormats.FileDrop);
        foreach (var path in files)
            ProcessaFile(path);
    }

    private void RichTextBox1_DragEnter(object sender, Global.System.Windows.Forms.DragEventArgs e)
    {
        if (e.Data.GetDataPresent(DataFormats.FileDrop))
        {
            e.Effect = DragDropEffects.Copy;
        }
    }

    public void ProcessaFile(string path)
    {
        this.RichTextBox1.AppendText(path + Environment.NewLine);
    }

    private void Button1_Click(object sender, EventArgs e)
    {
        var tfp = new TextFieldParser(this.RichTextBox1.Text.Trim);
        tfp.Delimiters = new string[] { "," };
        tfp.TextFieldType = FieldType.Delimited;
        string desc = tfp.ReadLine();
        this.RichTextBox2.AppendText(desc + Environment.NewLine);
        this.RichTextBox2.AppendText("__________________________________________________________" + Environment.NewLine + Environment.NewLine);
        while (tfp.EndOfData == false)
        {
            var fields = tfp.ReadFields();
            var r = new Ramen();
            r.numberOfReview = Conversions.ToInteger(fields[0]);
            r.brand = fields[1];
            r.variety = fields[2];
            r.style = fields[3];
            if (fields[4] is object)
            {
                r.country = fields[4];
            }

            if (string.Equals(fields[5], "Unrated"))
            {
                r.stars = (int)Math.Round(0.0);
            }
            else
            {
                r.stars = Conversions.ToInteger(fields[5]);
                // Me.RichTextBox2.AppendText(fields(5) & Environment.NewLine)
            }

            this.RichTextBox2.AppendText(string.Format("{0} - {1} - {2} - {3} - {4} - {5} ", r.numberOfReview, r.brand, r.variety, r.style, r.country, r.stars) + Environment.NewLine);
            if (r is object)
            {
                listOfValues.Add(r);
            }
        }

        Button3.Enabled = true;
    }

    private void ButtonGetFile_Click(object sender, EventArgs e)
    {
        var ofd = new OpenFileDialog();
        ofd.ShowDialog();
        if (ofd.FileName is object)
        {
            this.RichTextBox1.Text = ofd.FileName;
        }
    }

    private void Button3_Click(object sender, EventArgs e)
    {
        this.RichTextBox3.AppendText("ARITHMETIC MEAN OF # OF REVIEWS " + Environment.NewLine);
        this.RichTextBox3.AppendText("________________________________" + Environment.NewLine);
        int totalNumber = 2580;
        int sum = 0;
        foreach (var ramen in listOfValues)
            sum += ramen.numberOfReview;
        double average = Math.Round(sum / (double)(totalNumber - 1), 2);
        this.RichTextBox3.AppendText(average + Environment.NewLine);
        this.RichTextBox3.AppendText("DISTRIBUTION OF RAMEN SHOPS IN COUNTRIES" + Environment.NewLine);
        this.RichTextBox3.AppendText("________________________________" + Environment.NewLine);
        var frequencyDist = new Dictionary<string, Frequencies>();
        foreach (Ramen r in listOfValues)
        {
            if (frequencyDist.ContainsKey(r.country))
            {
                frequencyDist[r.country].count += 1m;
            }
            else
            {
                frequencyDist.Add(r.country, new Frequencies(1m));
            }

            frequencyDist[r.country].relativeFreq = Math.Round(frequencyDist[r.country].count / (totalNumber - 1), 2);
            frequencyDist[r.country].precentage = frequencyDist[r.country].relativeFreq * 100m;
        }

        this.RichTextBox3.AppendText("Country ".PadRight(12) + "Frequency ".PadRight(12) + "RFreq ".PadRight(12) + "Percentage ".PadRight(12) + Environment.NewLine);
        foreach (KeyValuePair<string, Frequencies> kvp in frequencyDist)
            this.RichTextBox3.AppendText(kvp.Key.ToString().PadRight(12) + " " + kvp.Value.count.ToString().PadRight(12) + " " + kvp.Value.relativeFreq.ToString().PadRight(12) + " " + kvp.Value.precentage.ToString() + " %" + Environment.NewLine);
        this.RichTextBox3.AppendText(Environment.NewLine + "DISTRIBUTION OF RATINGS OF RAMEN SHOPS IN JAPAN" + Environment.NewLine);
        this.RichTextBox3.AppendText("________________________________" + Environment.NewLine);
        var ratingsDist = new Dictionary<int, Frequencies>();
        ratingsDist.Add(5, new Frequencies(0m));
        ratingsDist.Add(4, new Frequencies(0m));
        ratingsDist.Add(3, new Frequencies(0m));
        ratingsDist.Add(2, new Frequencies(0m));
        ratingsDist.Add(1, new Frequencies(0m));
        int totalNum = 0;
        foreach (Ramen r in listOfValues)
        {
            if (r.country.Equals("Japan"))
            {
                if (r.stars == 5)
                {
                    ratingsDist[5].count += 1m;
                    totalNum += 1;
                }

                if (r.stars == 4)
                {
                    ratingsDist[4].count += 1m;
                    totalNum += 1;
                }

                if (r.stars == 3)
                {
                    ratingsDist[3].count += 1m;
                    totalNum += 1;
                }

                if (r.stars == 2)
                {
                    ratingsDist[2].count += 1m;
                    totalNum += 1;
                }

                if (r.stars == 1)
                {
                    ratingsDist[1].count += 1m;
                    totalNum += 1;
                }
            }
        }

        ratingsDist[5].relativeFreq = Math.Round(ratingsDist[5].count / (totalNum - 1), 2);
        ratingsDist[5].precentage = ratingsDist[5].relativeFreq * 100m;
        ratingsDist[4].relativeFreq = Math.Round(ratingsDist[4].count / (totalNum - 1), 2);
        ratingsDist[4].precentage = ratingsDist[4].relativeFreq * 100m;
        ratingsDist[3].relativeFreq = Math.Round(ratingsDist[3].count / (totalNum - 1), 2);
        ratingsDist[3].precentage = ratingsDist[3].relativeFreq * 100m;
        ratingsDist[2].relativeFreq = Math.Round(ratingsDist[2].count / (totalNum - 1), 2);
        ratingsDist[2].precentage = ratingsDist[2].relativeFreq * 100m;
        ratingsDist[1].relativeFreq = Math.Round(ratingsDist[1].count / (totalNum - 1), 2);
        ratingsDist[1].precentage = ratingsDist[1].relativeFreq * 100m;
        this.RichTextBox3.AppendText("Rating ".PadRight(12) + "Frequency ".PadRight(12) + "RFreq ".PadRight(12) + "Percentage ".PadRight(12) + Environment.NewLine);
        foreach (KeyValuePair<int, Frequencies> kvp in ratingsDist)
            this.RichTextBox3.AppendText(kvp.Key.ToString().PadRight(12) + " " + kvp.Value.count.ToString().PadRight(12) + " " + kvp.Value.relativeFreq.ToString().PadRight(12) + " " + kvp.Value.precentage.ToString() + " %" + Environment.NewLine);
    }
}

public partial class Ramen
{
    public string brand, variety, style, country;
    public int numberOfReview;
    public int stars;
}

public partial class Frequencies
{
    public decimal count;
    public decimal relativeFreq;
    public decimal precentage;

    public Frequencies(decimal c)
    {
        count = c;
    }

    public Frequencies(decimal c, decimal rf, decimal p)
    {
        count = c;
        relativeFreq = rf;
        precentage = p;
    }
}
