Intro⌗

in this assignment, I had to code a demonstrative program which computes the online arithmetic mean (if it’s a numeric variable) and an algo to compute the distribution for a discrete variable and for a continuous variable (can use values simulated with RANDOM object).

C# Code:⌗ 

using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace _2_A
{
    public partial class Form1 : Form
    {
        public Random R = new Random();

        public Form1()
        {
            InitializeComponent();
        }

        private void button1_Click(object sender, EventArgs e)
        {
            this.richTextBox1.Clear();

            int studentNumber = 101;
            int minHeight = 140;
            int maxHeight = 200;

            List<Student> listOfStudents = new List<Student>(studentNumber - 1);

            this.richTextBox1.AppendText("DATASET - Cryptography exam : Grades of students" + Environment.NewLine + Environment.NewLine);
            this.richTextBox1.AppendText("Total Count: " + (studentNumber - 1).ToString() + Environment.NewLine + Environment.NewLine);

            for (int i = 1; i < studentNumber; i++)
            {
                int randomNumber = R.Next(18, 31);

                decimal randomHeight = Math.Round((decimal)R.Next(minHeight, maxHeight), 2);

                this.richTextBox1.AppendText(("Student " + i.ToString() + ": ").PadRight(15) + randomNumber.ToString().PadRight(15) + randomHeight.ToString() + Environment.NewLine);

                Student s = new Student(randomNumber,randomHeight);

                listOfStudents.Add(s);
            }

            // OFFLINE ALGO

            decimal sum = 0;
            
            foreach (Student s in listOfStudents)
            {
                sum += s.gradeForCryptography;
            }

            decimal average = Math.Round(sum / (studentNumber - 1),2);

            // ONLINE ALGO

            double currentAvg = 0;
            int index = 0;

            foreach (Student s in listOfStudents)
            {
                index += 1;
                currentAvg = currentAvg + (s.gradeForCryptography - currentAvg) / index;
            }

            this.richTextBox1.AppendText("___________________________________" + Environment.NewLine);

            this.richTextBox1.AppendText("Arithmetic mean for Grades(naive formula): " + average.ToString("F") + Environment.NewLine);
            this.richTextBox1.AppendText("Arithmetic mean for Grades(online formula): " + currentAvg.ToString("F") + Environment.NewLine);

            Dictionary<int, Frequencies> frequencyDist = new Dictionary<int,Frequencies>();

            foreach (Student s in listOfStudents)
            {
                if (frequencyDist.ContainsKey(s.gradeForCryptography))
                {
                    frequencyDist[s.gradeForCryptography].count += 1;
                }
                else
                {
                    frequencyDist.Add(s.gradeForCryptography, new Frequencies(1));
                }

                frequencyDist[s.gradeForCryptography].relativeFreq = Math.Round(frequencyDist[s.gradeForCryptography].count / (studentNumber - 1),2);
                frequencyDist[s.gradeForCryptography].precentage = frequencyDist[s.gradeForCryptography].relativeFreq * 100;
            }

            this.richTextBox1.AppendText("___________________________________" + Environment.NewLine);
            this.richTextBox1.AppendText("Grade ".PadRight(7) + "Frequency " .PadRight(7) + "RFreq ".PadRight(7) + "Percentage ".PadRight(7) + Environment.NewLine);

            foreach (KeyValuePair<int,Frequencies> kvp in frequencyDist)
            {
                this.richTextBox1.AppendText(kvp.Key.ToString().PadRight(7) + " " + kvp.Value.count.ToString().PadRight(7) + " " + kvp.Value.relativeFreq.ToString().PadRight(7) + " " + kvp.Value.precentage.ToString() + " %" + Environment.NewLine);
            }

            decimal intervalSize = 10;
            decimal startingEndPoint = 170;

            //first interval

            Interval myInterval = new Interval();
            myInterval.lowerEnd = startingEndPoint;
            myInterval.upperEnd = startingEndPoint + intervalSize;

            List<Interval> listOfIntervals = new List<Interval>();

            listOfIntervals.Add(myInterval);

            foreach(Student s in listOfStudents)
            {
                Boolean studentInserted = false;

                foreach(Interval i in listOfIntervals)
                {
                    if (s.height > i.lowerEnd && s.height <= i.upperEnd)
                    {
                        i.count += 1;
                        studentInserted = true;
                    }
                }

                if(studentInserted == true)
                {
                    continue;
                }

                if(s.height <= listOfIntervals[0].lowerEnd)
                {
                    while (true)
                    {
                        Interval newLeftInterval = new Interval();
                        newLeftInterval.upperEnd = listOfIntervals[0].lowerEnd;
                        newLeftInterval.lowerEnd = newLeftInterval.upperEnd - intervalSize;

                        listOfIntervals.Insert(0, newLeftInterval);

                        if (s.height > newLeftInterval.lowerEnd && s.height <= newLeftInterval.upperEnd)
                        {
                            newLeftInterval.count += 1;
                            studentInserted = true;
                            break;
                        }

                    }
                }

                else if(s.height > listOfIntervals[listOfIntervals.Count - 1].upperEnd)
                {
                    while (true)
                    {
                        Interval newRightInterval = new Interval();
                        newRightInterval.lowerEnd = listOfIntervals[listOfIntervals.Count - 1].upperEnd;
                        newRightInterval.upperEnd = newRightInterval.lowerEnd + intervalSize;

                        listOfIntervals.Add(newRightInterval);

                        if (s.height > newRightInterval.lowerEnd && s.height <= newRightInterval.upperEnd)
                        {

                            newRightInterval.count += 1;
                            studentInserted = true;
                            break;
                        }

                    }

                }
            }

            this.richTextBox1.AppendText("___________________________________" + Environment.NewLine);
            this.richTextBox1.AppendText("HEIGHT DISTRIBUTION" + Environment.NewLine);
            this.richTextBox1.AppendText("Interval".PadRight(20) + "Count" + Environment.NewLine);

            foreach(Interval i in listOfIntervals)
            {
                this.richTextBox1.AppendText("[ " + i.lowerEnd + " " + i.upperEnd + "] ".PadRight(11) + i.count + Environment.NewLine);
            }
        }

        public class Frequencies
        {
            public decimal count;
            public decimal relativeFreq;
            public decimal precentage;

            public Frequencies(decimal c)
            {
                this.count = c;
            }
            public Frequencies(decimal c, decimal rf, decimal p)
            {
                this.count = c;
                this.relativeFreq = rf;
                this.precentage = p;
            }
        }

        public class Interval
        {
            public decimal lowerEnd, upperEnd, count, relativeFrequency, percentage;

        }
    }
}

            For Each s As Student In listOfStudents

                For Each i As Interval In listOfIntervals

                    If s.height > i.lowerEnd AndAlso s.height <= i.upperEnd Then

                            newLeftInterval.count += 1
                            studentInserted = True
                            Exit While

                            newRightInterval.count += 1
                            studentInserted = True
                            Exit While
                        End If
